public static void GenerateSalesSummary(IEnumerable<string> salesFiles, string reportFile)
    {
        double totalSales = 0;
        var details = new List<string>();

        foreach (var file in salesFiles)
        {
            string salesJson = File.ReadAllText(file);
            SalesData? data = JsonConvert.DeserializeObject<SalesData?>(salesJson);
            double fileTotal = data?.Total ?? 0;

            totalSales += fileTotal;
            details.Add($"  {Path.GetFileName(file)}: ${fileTotal:N2}");
        }

        var reportContent = $"Sales Summary\n" +
                            $"----------------------------\n" +
                            $" Total Sales: ${totalSales:N2}\n\n" +
                            $" Details:\n{string.Join(Environment.NewLine, details)}";

        File.WriteAllText(reportFile, reportContent);
    }
